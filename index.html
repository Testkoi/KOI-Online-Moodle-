function selectCourse(courseId) { const courseMap = { '1': 'CourseDetails.html?course=1', '2': 'CourseDetails.html?course=2', '3': 'CourseDetails.html?course=3', '4': 'CourseDetails.html?course=4' }; if (courseMap[courseId]) { window.location.href = courseMap[courseId]; } else { console.error('Course ID not found:', courseId); } }
